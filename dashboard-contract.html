function updateChart() { if (!currentContract) return; const sections = currentContract.contract_sections_breakdown || {}; const start = new Date(currentContract.contract_start); const end = new Date(currentContract.contract_end); const today = new Date(); const totalMonths = Math.max(1, (end.getFullYear() - start.getFullYear()) * 12 + (end.getMonth() - start.getMonth())); const monthsElapsed = Math.max(1, (today.getFullYear() - start.getFullYear()) * 12 + (today.getMonth() - start.getMonth())); let labels = [], targetAvgData = [], actualAvgData = []; selectedSections.sort().forEach(s => { labels.push(s); const fullBudget = sections[s]?.amt || 0; const targetMonthly = fullBudget / totalMonths; targetAvgData.push(targetMonthly.toFixed(2)); let totalUsed = 0; currentWO.forEach(wo => { if(wo.section_category?.[s]) totalUsed += wo.section_category[s].amt || 0; }); const actualMonthly = totalUsed / monthsElapsed; actualAvgData.push(actualMonthly.toFixed(2)); }); if (myChart) { myChart.data.labels = labels; myChart.data.datasets[0].data = targetAvgData; myChart.data.datasets[1].data = actualAvgData; myChart.update(); } else { const ctx = document.getElementById('healthChart').getContext('2d'); myChart = new Chart(ctx, { type: 'bar', data: { labels: labels, datasets: [ { label: 'Anggaran Bulanan', data: targetAvgData, backgroundColor: 'rgba(200, 200, 205, 0.5)', borderRadius: 5 }, { label: 'Sebenar Bulanan', data: actualAvgData, backgroundColor: '#007AFF', borderRadius: 5 } ] }, options: { responsive: true, maintainAspectRatio: false, layout: { padding: { bottom: 10 // Beri ruang bawah supaya label tak terpotong } }, plugins: { legend: { position: 'top', labels: { font: { size: 10, weight: 'bold' } } }, datalabels: { anchor: 'end', align: 'top', font: { size: 8, weight: 'bold' }, formatter: (val) => val > 0 ? `RM${Math.round(val).toLocaleString()}` : '' } }, scales: { y: { beginAtZero: true, ticks: { font: { size: 9 } } }, x: { display: true, grid: { display: false }, ticks: { display: true, autoSkip: false, // PAKSA semua section keluar maxRotation: 45, // Sengetkan jika banyak sangat section minRotation: 45, font: { size: 10, weight: 'bold' }, color: '#666' } } } } }); } }
