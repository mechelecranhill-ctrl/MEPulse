function updateChart() { const sections = currentContract.contract_sections_breakdown || {}; let labels = [], bData = [], sData = []; selectedSections.sort().forEach(s => { labels.push(s); const budget = sections[s].amt || 0; bData.push(budget); let used = 0; currentWO.forEach(wo => { if(wo.section_category?.[s]) used += wo.section_category[s].amt || 0; }); sData.push(used); }); if (myChart) { myChart.data.labels = labels; myChart.data.datasets[0].data = bData; myChart.data.datasets[1].data = sData; myChart.update(); } else { const ctx = document.getElementById('healthChart').getContext('2d'); myChart = new Chart(ctx, { type: 'bar', data: { labels: labels, datasets: [ { label: 'Bajet', data: bData, backgroundColor: '#e5e5ea', borderRadius: 8 }, { label: 'Guna', data: sData, backgroundColor: '#007AFF', borderRadius: 8 } ] }, options: { responsive: true, maintainAspectRatio: false, animation: { duration: 800, easing: 'easeOutQuart' }, plugins: { datalabels: { align: 'top', anchor: 'end', textAlign: 'center', formatter: (value, context) => { // Hanya papar peratusan pada dataset 'Guna' (index 1) if (context.datasetIndex === 1) { const budget = context.chart.data.datasets[0].data[context.dataIndex]; const percent = budget > 0 ? Math.round((value / budget) * 100) : 0; return value > 0 ? `RM ${value.toLocaleString()}\n(${percent}%)` : ''; } // Untuk dataset 'Bajet' (index 0), papar RM sahaja return value > 0 ? `RM ${value.toLocaleString()}` : ''; }, font: { size: 9, weight: 'bold' }, color: (context) => context.datasetIndex === 1 ? '#007AFF' : '#8e8e93' } }, scales: { y: { beginAtZero: true, grid: { display: false } } } } }); } }
